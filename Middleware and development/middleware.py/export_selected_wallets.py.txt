import json

def export_wallets():
    """
    Export eligible wallets for use in the reward distribution process.
    """
    with open("data/eligible_wallets.json", "r") as f:
        eligible_wallets = json.load(f)

    # Here, you can add logic to format/export wallets to Chainlink VRF
    with open("data/exported_wallets.json", "w") as f:
        json.dump(eligible_wallets, f)
    print("Wallets exported to Chainlink VRF.")

if __name__ == "__main__":
    export_wallets()
